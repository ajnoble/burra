# Phase 14: Booking Engine Rules Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Auto-cancel unpaid bookings after a configurable deadline + grace period, with email reminders before the deadline.

**Architecture:** Daily cron job queries CONFIRMED unpaid bookings, sends reminders based on configurable day thresholds, then auto-cancels bookings past the grace period. Config is set at org level (defaults) with per-round overrides and per-booking admin overrides. Reuses existing `cancelBooking()` for the actual cancellation.

**Tech Stack:** Next.js 16 App Router, Drizzle ORM, PostgreSQL, Vitest, React Email + Resend, Stripe Connect (refunds)

**Note:** Booking rounds have no admin CRUD UI — they are managed via seed scripts and direct DB. The round-level schema columns (`balanceDueDate`, `paymentGraceDays`, etc.) are added but no round form UI is built. Admins configure defaults at org level and override per-booking.

import { db } from "@/db/index";
import {
  bookings,
  bookingRounds,
  organisations,
  members,
  lodges,
  bedHolds,
} from "@/db/schema";
import { and, eq, isNull, isNotNull, lt } from "drizzle-orm";
import { sendEmail } from "@/lib/email/send";
import React from "react";
import { BookingPaymentReminderEmail } from "@/lib/email/templates/booking-payment-reminder";
import { BookingAutoCancelledEmail } from "@/lib/email/templates/booking-auto-cancelled";
import { AdminBookingNotificationEmail } from "@/lib/email/templates/admin-booking-notification";
import { cancelBooking } from "./cancel";

type CronResult = {
  remindersSent: number;
  bookingsCancelled: number;
  holdsCleared: boolean;
};

type UnpaidBookingRow = {
  bookingId: string;
  bookingReference: string;
  lodgeName: string;
  lodgeId: string;
  checkInDate: string;
  checkOutDate: string;
  totalAmountCents: number;
  balanceDueDate: string;
  paymentRemindersSentAt: Record<string, string> | null;
  cancellationPolicyId: string | null;
  primaryMemberId: string;
  organisationId: string;
  roundPaymentReminderDays: number[] | null;
  roundPaymentGraceDays: number | null;
  roundAutoCancelRefundPolicy: string | null;
  orgPaymentReminderDays: number[];
  orgPaymentGraceDays: number;
  orgName: string;
  orgSlug: string;
  orgContactEmail: string | null;
  orgLogoUrl: string | null;
  memberEmail: string;
  memberFirstName: string;
  memberLastName: string;
};

function daysUntil(dateStr: string): number {
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  const target = new Date(dateStr + "T00:00:00");
  return Math.floor((target.getTime() - today.getTime()) / (1000 * 60 * 60 * 24));
}

function getUnpaidBookingsQuery() {
  return db
    .select({
      bookingId: bookings.id,
      bookingReference: bookings.bookingReference,
      lodgeName: lodges.name,
      lodgeId: bookings.lodgeId,
      checkInDate: bookings.checkInDate,
      checkOutDate: bookings.checkOutDate,
      totalAmountCents: bookings.totalAmountCents,
      balanceDueDate: bookings.balanceDueDate,
      paymentRemindersSentAt: bookings.paymentRemindersSentAt,
      cancellationPolicyId: bookings.cancellationPolicyId,
      primaryMemberId: bookings.primaryMemberId,
      organisationId: bookings.organisationId,
      roundPaymentReminderDays: bookingRounds.paymentReminderDays,
      roundPaymentGraceDays: bookingRounds.paymentGraceDays,
      roundAutoCancelRefundPolicy: bookingRounds.autoCancelRefundPolicy,
      orgPaymentReminderDays: organisations.bookingPaymentReminderDays,
      orgPaymentGraceDays: organisations.bookingPaymentGraceDays,
      orgName: organisations.name,
      orgSlug: organisations.slug,
      orgContactEmail: organisations.contactEmail,
      orgLogoUrl: organisations.logoUrl,
      memberEmail: members.email,
      memberFirstName: members.firstName,
      memberLastName: members.lastName,
    })
    .from(bookings)
    .innerJoin(lodges, eq(lodges.id, bookings.lodgeId))
    .innerJoin(bookingRounds, eq(bookingRounds.id, bookings.bookingRoundId))
    .innerJoin(organisations, eq(organisations.id, bookings.organisationId))
    .innerJoin(members, eq(members.id, bookings.primaryMemberId))
    .where(
      and(
        eq(bookings.status, "CONFIRMED"),
        isNull(bookings.balancePaidAt),
        isNotNull(bookings.balanceDueDate)
      )
    );
}

async function processReminders(rows: UnpaidBookingRow[]): Promise<number> {
  const appUrl = process.env.NEXT_PUBLIC_APP_URL || "http://localhost:3000";
  let remindersSent = 0;

  for (const row of rows) {
    const reminderDays = row.roundPaymentReminderDays ?? row.orgPaymentReminderDays;
    const alreadySent = row.paymentRemindersSentAt ?? {};
    const daysRemaining = daysUntil(row.balanceDueDate);

    // Sort descending — check largest threshold first
    const sorted = [...reminderDays].sort((a, b) => b - a);

    for (const dayThreshold of sorted) {
      const key = String(dayThreshold);
      if (alreadySent[key]) continue;
      if (daysRemaining > dayThreshold) continue;

      // Send reminder
      sendEmail({
        to: row.memberEmail,
        subject: `Payment reminder — ${row.bookingReference}`,
        template: React.createElement(BookingPaymentReminderEmail, {
          orgName: row.orgName,
          bookingReference: row.bookingReference,
          lodgeName: row.lodgeName,
          checkInDate: row.checkInDate,
          checkOutDate: row.checkOutDate,
          totalAmountCents: row.totalAmountCents,
          balanceDueDate: row.balanceDueDate,
          daysRemaining: Math.max(0, daysRemaining),
          payUrl: `${appUrl}/${row.orgSlug}/dashboard`,
          logoUrl: row.orgLogoUrl || undefined,
        }),
        replyTo: row.orgContactEmail || undefined,
        orgName: row.orgName,
      });

      const updatedSentAt = { ...alreadySent, [key]: new Date().toISOString() };
      await db
        .update(bookings)
        .set({ paymentRemindersSentAt: updatedSentAt, updatedAt: new Date() })
        .where(eq(bookings.id, row.bookingId));

      remindersSent++;
      break; // Only send one reminder per booking per cron run
    }
  }

  return remindersSent;
}

async function processAutoCancels(rows: UnpaidBookingRow[]): Promise<number> {
  const appUrl = process.env.NEXT_PUBLIC_APP_URL || "http://localhost:3000";
  let bookingsCancelled = 0;

  for (const row of rows) {
    const graceDays = row.roundPaymentGraceDays ?? row.orgPaymentGraceDays;
    const daysPastDue = -daysUntil(row.balanceDueDate);

    if (daysPastDue < graceDays) continue;

    // Determine refund policy
    const refundPolicy = row.roundAutoCancelRefundPolicy ?? "cancellation_policy";
    let refundOverrideCents: number | undefined;

    if (refundPolicy === "none") {
      refundOverrideCents = 0;
    } else if (refundPolicy === "full") {
      refundOverrideCents = row.totalAmountCents;
    }
    // "cancellation_policy" → leave refundOverrideCents undefined, cancelBooking handles it

    const cancelResult = await cancelBooking({
      bookingId: row.bookingId,
      organisationId: row.organisationId,
      cancelledByMemberId: row.primaryMemberId,
      reason: "Auto-cancelled: payment deadline expired",
      refundOverrideCents,
      slug: row.orgSlug,
    });

    if (cancelResult.success) {
      // Send auto-cancel specific email (in addition to the standard cancel email)
      sendEmail({
        to: row.memberEmail,
        subject: `Booking auto-cancelled — ${row.bookingReference}`,
        template: React.createElement(BookingAutoCancelledEmail, {
          orgName: row.orgName,
          bookingReference: row.bookingReference,
          lodgeName: row.lodgeName,
          checkInDate: row.checkInDate,
          checkOutDate: row.checkOutDate,
          totalAmountCents: row.totalAmountCents,
          refundAmountCents: cancelResult.refundAmountCents,
          logoUrl: row.orgLogoUrl || undefined,
        }),
        replyTo: row.orgContactEmail || undefined,
        orgName: row.orgName,
      });

      // Send admin notification
      if (row.orgContactEmail) {
        sendEmail({
          to: row.orgContactEmail,
          subject: `[Admin] Booking auto-cancelled — ${row.bookingReference}`,
          template: React.createElement(AdminBookingNotificationEmail, {
            orgName: row.orgName,
            bookingReference: row.bookingReference,
            memberName: `${row.memberFirstName} ${row.memberLastName}`,
            lodgeName: row.lodgeName,
            checkInDate: row.checkInDate,
            checkOutDate: row.checkOutDate,
            action: "cancelled" as const,
            adminUrl: `${appUrl}/${row.orgSlug}/admin/bookings/${row.bookingId}`,
            logoUrl: row.orgLogoUrl || undefined,
          }),
          orgName: row.orgName,
        });
      }

      bookingsCancelled++;
    }
  }

  return bookingsCancelled;
}

export async function processBookingPaymentCron(): Promise<CronResult> {
  // Pass 1: Send payment reminders
  const unpaidBookings = await getUnpaidBookingsQuery() as UnpaidBookingRow[];
  const remindersSent = await processReminders(unpaidBookings);

  // Pass 2: Auto-cancel overdue bookings (re-query to exclude any just-paid)
  const overdueBookings = await getUnpaidBookingsQuery() as UnpaidBookingRow[];
  const bookingsCancelled = await processAutoCancels(overdueBookings);

  // Pass 3: Clean up expired bed holds
  await db.delete(bedHolds).where(lt(bedHolds.expiresAt, new Date()));

  return { remindersSent, bookingsCancelled, holdsCleared: true };
}

export async function POST(request: NextRequest): Promise<Response> {
  const authHeader = request.headers.get("authorization");
  const cronSecret = process.env.CRON_SECRET;

    holdsCleared: result.holdsCleared,
  });
}

- [ ] **Step 3: Create the server action**

Create `src/actions/bookings/update-due-date.ts`:

```typescript
"use server";

import { db } from "@/db/index";
import { bookings } from "@/db/schema";
import { and, eq } from "drizzle-orm";
import { revalidatePath } from "next/cache";
import { getSessionMember, canAccessAdmin } from "@/lib/auth";

type UpdateDueDateInput = {
  bookingId: string;
  organisationId: string;
  balanceDueDate: string | null;
  slug: string;
};

export async function updateBookingDueDate(
  input: UpdateDueDateInput
): Promise<{ success: boolean; error?: string }> {
  const session = await getSessionMember(input.organisationId);
  if (!session || !canAccessAdmin(session.role)) {
    return { success: false, error: "Not authorised" };
  }

  await db
    .update(bookings)
    .set({
      balanceDueDate: input.balanceDueDate,
      updatedAt: new Date(),
    })
    .where(
      and(
        eq(bookings.id, input.bookingId),
        eq(bookings.organisationId, input.organisationId)
      )
    );

  revalidatePath(`/${input.slug}/admin/bookings/${input.bookingId}`);

  return { success: true };
}
```

```typescript
"use client";

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { updateBookingDueDate } from "@/actions/bookings/update-due-date";
import { toast } from "sonner";

type Props = {
  bookingId: string;
  organisationId: string;
  currentDueDate: string | null;
  slug: string;
};

export function EditDueDateForm({
  bookingId,
  organisationId,
  currentDueDate,
  slug,
}: Props) {
  const [editing, setEditing] = useState(false);
  const [saving, setSaving] = useState(false);
  const [dueDate, setDueDate] = useState(currentDueDate ?? "");

  async function handleSave() {
    setSaving(true);
    try {
      const result = await updateBookingDueDate({
        bookingId,
        organisationId,
        balanceDueDate: dueDate || null,
        slug,
      });
      if (result.success) {
        toast.success("Payment due date updated");
        setEditing(false);
      } else {
        toast.error(result.error ?? "Failed to update");
      }
    } catch {
      toast.error("Failed to update");
    } finally {
      setSaving(false);
    }
  }

  if (!editing) {
    return (
      <Button variant="ghost" size="sm" onClick={() => setEditing(true)}>
        {currentDueDate ? "Edit" : "Set due date"}
      </Button>
    );
  }

  return (
    <div className="flex items-center gap-2">
      <Input
        type="date"
        value={dueDate}
        onChange={(e) => setDueDate(e.target.value)}
        className="w-40"
      />
      <Button size="sm" onClick={handleSave} disabled={saving}>
        {saving ? "Saving..." : "Save"}
      </Button>
      <Button
        size="sm"
        variant="ghost"
        onClick={() => {
          setDueDate(currentDueDate ?? "");
          setEditing(false);
        }}
      >
        Cancel
      </Button>
    </div>
  );
}
```
